        const randomPassword = () => {
          const symbols = '!@#$%^&*()_-+={}[];:<>,.?/~';
          const allChars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz1234567890' + symbols;
          let password = '';
          for (let i = 0; i < 10; i++) {
            const randomIndex = Math.floor(Math.random() * allChars.length);
            password += allChars[randomIndex];
          }
          return password;
        }
